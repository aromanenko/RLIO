N - id файла расположения
"MERGE_TABLE_STORE_N.csv":
	- product_id: id текущего товара
	- store_id: id текущего расположения
	- curr_date: текущая дата
	- s_qty: продажи, кол-во уп.
	- flg_spromo: факт наличия супер-промо (1: было, 0: не было)
	- stock: текущий остаток на сегодняший день, кол-во уп.

"echelon_1_batch_size.csv"
	- date_from: дата начала интервала
	- date_to: дата окончание интервала (включительно)
	- product_id: id текущего товара
	- location_id: id текущего расположения
	- value: значение batch_size (кратность заказа, количество)

	- location_id: id текущего расположения
	- value: значение lt (срок доставки)
